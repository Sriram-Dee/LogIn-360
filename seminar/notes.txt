									TRANSITION
 
 
 transition-property: ; 
 
			Specifies the name of the CSS property the transition effect is for
        ( height,width, background-color, border-radius ... ect )
		

 transition-duration: ;
            Specifies how many seconds or milliseconds a transition effect takes to complete 


 transition-delay: ; 
			Specifies a delay (in seconds) for the transition effect 
			

 transition-timing-function: ; 
			Specifies the speed curve of the transition effect
            ease(default), 
			ease-in, 
			ease-in-out,
			ease-out 
			linear,
			cubic-bezier(s,s,s,s) 
			

 transition: ; 
	 A shorthand property for setting the four transition properties into a single property
            (transition-property  transition-duration  delay   timing-function)

									ANIMATION


	1	animation-name:; (our preferance)
				Specifies the name of the keyframe you want to bind
				
	2	animation-duration:;  ( in seconds or milliseconds 10s or 3000ms )
				Specifies how many seconds or milliseconds an animation takes to complete
				
				
	3	animation-timing-function:; (linear,ease,ease-in,ease-out,ease-in-out,cubic-bezier(n,n,n,n))
				Specifies the speed curve of the animation
		
	4	animation-delay:; ( in seconds or milliseconds 3s or 3000ms )
				Specifies a delay before the animation will start
				
	5	animation-iteration-count:; (number,infinite)
				Specifies how many times an animation should be played
				
	6	animation-direction: ; (normal, reverse, alternate, alternate-reverse)
				Specifies whether or not the animation should play in reverse on alternate cycles
				
				normal	(Default value)
					The animation is played as normal (forwards)
					
				reverse	
					The animation is played in reverse direction (backwards)
					
				alternate	
					The animation is played forwards first, then backwards	
				
				alternate-reverse
					The animation is played backwards first, then forwards
					
	7    animation: ;(name duration timing-function delay iteration-count direction)
					Short-hand
